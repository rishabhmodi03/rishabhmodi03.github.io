def analyze_data(financial_records):
"""
Analyze financial data using Python
"""
import pandas as pd
import numpy as np

# Convert to DataFrame
df = pd.DataFrame(financial_records)

# Calculate key metrics
summary = {
    "total_revenue": df["revenue"].sum(),
    "average_expense": df["expenses"].mean(),
    "profit_margin": (df["revenue"].sum() - df["expenses"].sum()) / df["revenue"].sum(),
    "growth_rate": (df["revenue"].iloc[-1] - df["revenue"].iloc[0]) / df["revenue"].iloc[0]
}

return summary
